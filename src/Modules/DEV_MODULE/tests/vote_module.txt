#vote_module
/text Start VOTE_MODULE
!poll
/delay %1
!poll kill error
/delay %1
!vote remove error
/delay %1
!poll end error
/delay %1
!poll end 1
/delay %1
!poll create 1h test|option1|option2|option3
/delay %1
!vote 1 error|option4
/delay %1
!vote 1 option4
/delay %1
!vote remove 1
/delay %1
!vote 1 option3
/delay %1
!poll end test
/delay %1
!poll kill test
/delay %1
/text End VOTE_MODULE
